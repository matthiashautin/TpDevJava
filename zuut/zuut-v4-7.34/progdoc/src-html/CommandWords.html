001/**
002 * This class is part of the "World of Zuul" application. 
003 * "World of Zuul" is a very simple, text based adventure game.  
004 * 
005 * This class holds an enumeration table of all command words known to the game.
006 * It is used to recognise commands as they are typed in.
007 *
008 * @author  Michael Kolling and David J. Barnes + D.Bureau
009 * @version 2008.03.30 + 2019.09.25
010 */
011public class CommandWords
012{
013    // a constant array that will hold all valid command words
014    private static final String aValidCommands[] = {"go", "help", "quit", "look", "eat", "audio", "back", "test", "take", "drop", "infoplayer"};
015
016    /**
017     * Check whether a given String is a valid command word. 
018     * @return true if a given string is a valid command,
019     * false if it isn't.
020     */
021    public boolean isCommand( final String pString )
022    {
023        for ( int vI=0; vI<this.aValidCommands.length; vI++ ) {
024            if ( this.aValidCommands[vI].equals( pString ) )
025                return true;
026        } // for
027        // if we get here, the string was not found in the commands :
028        return false;
029    } // isCommand()
030
031    /**
032     * @return all commands that exist in aValidCommands() in a board
033     */
034    public String getCommandList() {
035        StringBuilder vCommands = new StringBuilder();
036
037        for(int i = 0; i < this.aValidCommands.length; i++) {            
038            vCommands.append( this.aValidCommands[i] + " | " );        
039        }        
040        return vCommands.toString();   
041
042    }//getCommandList()  
043
044} // CommandWords
