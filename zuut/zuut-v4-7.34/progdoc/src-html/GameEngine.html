148     * allows you to change room if this room exist c_a_d if the direction of the next room is not equal to null
149     * @param pCommand type Command allows the player to retrieve the print command
150     */
151    public void goRoom(final Command pCommand) {
152
153        if(! pCommand.hasSecondWord()) {//si le second mot n'existe pas 
154            this.aGui.println("Go Where ?");
155            return;
156        }
157
158        String vDirection = pCommand.getSecondWord();//initialise une variable de type String qui est le second mot de notre parametre
159        Room vNextRoom = this.aPlayer.getCurrentRoom().getExit(vDirection);//donne Ã  la variable vNextRoom la valeur de toutes les sorties qui existe pour la room actuelle
160
161        if (vNextRoom == null) {//si la room suivante n'a pas de direction associÃ©e
162            this.aGui.println("There is no door !");
163            return;
164        } else {//si on change de room on donne la valeur de aCurrentRoom a notre nouvelle room
165            this.aPlayer.getPreviousRoom().push(this.aPlayer.getCurrentRoom()); // mettre Ã  jour la piÃ¨ce prÃ©cÃ©dente 
166            this.aPlayer.setCurrentRoom(vNextRoom);
167            printLocationInfo();
168        }
169    } // goRoom()

194    /**
195     * analyse entered command and determines what action must be taken based on this commad and manages the different actions
196     * @param pAppelBonneMethod type Command represent the command entered by the player
197     **/
198    public void interpretCommand(final String pAppelBonneMethod) {
199        this.aGui.println( "> " + pAppelBonneMethod );
200        Command vCommand = this.aParser.getCommand( pAppelBonneMethod );
201
202        if(vCommand.isUnknown() ) { //si la Commande tapÃ©e est inconnu
203            this.aGui.println("I don't know what you mean...");
204            return;
205        } 
206
207        String vCommandWord = vCommand.getCommandWord();        
208
209        if(vCommandWord.equals("quit")) { //si la commande tapÃ©e est "quit"
210            if(vCommand.hasSecondWord() == true) {//si l'utilisateur tap un second mot aprÃ©s "quit" (exemple: "quit program")
211                this.aGui.println("Quit what ?");
212            } else {//sinon on quit 
213                this.endGame();
214            } 
215        }
216        else if(vCommandWord.equals("help")) { //si la commande tapÃ©e est "help"
217            this.printHelp();
218        } 
219        else if(vCommandWord.equals("go")) { //si la commande tapÃ©e est "go"
220            goRoom(vCommand);
221        }
222        else if(vCommandWord.equals("look")) { //si la commande tapÃ©e est "look"
223            look(vCommand);
224        }
225        else if(vCommandWord.equals("eat")) { //si la commande tapÃ©e est "eat"
226            eat(vCommand);
227        }
228        else if(vCommandWord.equals("audio")) { //si la commande tapÃ©e est "audio"
229            this.toggleAudio();
230
231        }else if(vCommandWord.equals("back")) { //si la commande tapÃ©e est "audio"
232            goback(vCommand);
233
234        }else if (vCommandWord.equals("test")) { //si la commande tapÃ©e est "test"
235            test(vCommand);
236
237        }else if (vCommandWord.equals("take")) { //si la commande tapÃ©e est "take"
238            take(vCommand);
239
240        }else if (vCommandWord.equals("drop")) { //si la commande tapÃ©e est "drop"
241            drop(vCommand);
242
243        }else if (vCommandWord.equals("infoplayer")) { //si la commande tapÃ©e est "infoplayer"
244            printPlayer();
245        }
246        else { //si print "Erreur du programmeur : commande non reconnue !"
247            this.aGui.println("Erreur du programmeur : commande non reconnue !");
248        }   
249    } //processCommand()

285    private void eat(final Command pSecondMot) {

287            String vItemName = pSecondMot.getSecondWord();
288            Item vItem = this.aPlayer.getItemInInventory(vItemName);
289
290            // Check if the item is in the player's inventory
291            if (vItem != null && vItem.getNameItem().equals("Cookie")) {
292                // Check if the player has the item in the inventory
293                if (this.aPlayer.getItemInInventory("Cookie") != null) {
294                    // Remove the Cookie from the player's inventory
295                    this.aPlayer.dropItem("Cookie", vItem);
296                    // Update player's health
297                    this.aPlayer.donneVie(20);
298                    this.aGui.println("You have eaten the Cookie. Your health is now " + this.aPlayer.getVie() + ".");
299                } else {
300                    this.aGui.println("You don't have a " + vItemName + " in your inventory.");
301                }
302            } else {
303                this.aGui.println("You can't eat that!");
304            }
305        } else {
306            this.aGui.println("Just one thing at a time.");
